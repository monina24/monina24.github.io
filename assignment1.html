The schema of the circuit. A 160 Ohm resistor is enough for this circuit, as found through Ohm's law. I decided to use a 1K resistor because even though it makes the LED more dim, it still allows the LED to light up and there's less of a chance of blowing out the LED.


// the setup function runs once when you press reset or power the board
void setup() {
  pinMode(LED_BUILTIN, OUTPUT);  // initialize digital pin LED_BUILTIN (red) as an output.
  pinMode(8, OUTPUT);            // initialize digital pin 8  (yellow) as an output.
  pinMode(7, OUTPUT);            // initialize digital pin 7 (red) as an output.
}

// the loop function runs over and over again forever
void loop() {
  digitalWrite(LED_BUILTIN, HIGH); // turn the red LED on (HIGH is the voltage level)
  digitalWrite(7, HIGH);           // turn the red LED on
  digitalWrite(8, LOW);            // turn the yellow LED off by making voltage LOW
  delay(500);                      // wait for .5 second
  digitalWrite(LED_BUILTIN, LOW);  // turn the red LED off  
  digitalWrite(7, LOW);            // turn the red LED off
  digitalWrite(8, HIGH);           // turn the yellow LED on
  delay(500);                      // wait for .5 second
}
